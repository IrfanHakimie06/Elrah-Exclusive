        function kiraHarga() {
            var itemElement = document.getElementById("item");
            var selectedItem = itemElement.value;
            var selectedPrice = parseInt(itemElement.options[itemElement.selectedIndex].getAttribute("data-price"));

            var selectedColor = document.getElementById("color").value;
            var selectedSize = document.getElementById("size").value;
            var quantity = parseInt(document.getElementById("quantity").value);
            var addSampin = document.getElementById("sampin").checked;
            
            if (quantity < 1 || quantity > 10) {
                alert("Kuantiti mesti antara 1 hingga 10!");
                return;
            }

            var sampinPrice = addSampin ? 79 : 0;
            var totalPrice = (selectedPrice + sampinPrice) * quantity;

            document.getElementById("receipt").innerHTML = `
                <strong>Item:</strong> ${selectedItem}<br>
                <strong>Warna:</strong> ${selectedColor}<br>
                <strong>Saiz:</strong> ${selectedSize}<br>
                <strong>Kuantiti:</strong> ${quantity}<br>
                <strong>Sampin:</strong> ${addSampin ? "Ya (+RM79)" : "Tidak"}<br>
                <h3><strong>Total Harga:</strong> RM${totalPrice}</h3>
            `;
        }
    </script>
